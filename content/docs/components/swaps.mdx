## Usage (Exact In)

```tsx showLineNumbers
<Swaps
  exactIn
  onComplete={completeCurrent}
  onError={handleError}
  onStart={handleStart}
  exactInprefill={{
    fromToken: "USDC",
    fromChainID: SUPPORTED_CHAINS.BASE,
    fromAmount: "100",
    toToken: "USDT",
    toChainID: SUPPORTED_CHAINS.ARBITRUM,
  }}
/>
```

## Usage (Exact Out)

```tsx showLineNumbers
<Swaps
  exactIn={false}
  onComplete={completeCurrent}
  onError={handleError}
  onStart={handleStart}
  exactOutprefill={{
    toToken: "USDT",
    toChainID: SUPPORTED_CHAINS.ARBITRUM,
    toAmount: "100",
  }}
/>
```
